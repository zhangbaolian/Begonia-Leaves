# 时间段，把黄帝纪年拆分为公元前和公元两个时间段，公元前和公元。
# 黄帝纪年2698年对应公元元年，也就是公元1年。
# 时间段
QHTY_SHIJIA_DUAN = {
	type = character
	text = {
		trigger = { qhty_shijian_xiaoyu < 2698 }
		localization_key = QHTY_SHIJIAN_GONGYUANQIAN
	}

	text = {
		trigger = { qhty_shijian_dayu_dengyu >= 2698 }
		localization_key = QHTY_TIANGAN_GONGYUAN
	}
}

# 年时间换算
QHTY_SHIJIAN_NIAN = {

	text = {

	}
